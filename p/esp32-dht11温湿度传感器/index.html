import socket  
import matplotlib.pyplot as plt  
import matplotlib.animation as animation  
from collections import deque  
  
# ESP32ÁöÑIPÂíåÁ´ØÂè£  
ESP32_IP = "192.168.31.35"  # ÊõøÊç¢‰∏∫ ESP32 IP Âú∞ÂùÄ  
PORT = 8080  # Á´ØÂè£Âè∑  
  
# ÂàõÂª∫SocketËøûÊé•

# Áª¥Êä§ÊúÄËøë 30 ‰∏™Êï∞ÊçÆÁÇπÔºàÊúÄÂ§ö 1 ÂàÜÈíüÔºâ  
MAX_POINTS = 30

temps = deque(maxlen=MAX_POINTS)  # Ê∏©Â∫¶Êï∞ÊçÆ

# ÂàõÂª∫ÂõæÂΩ¢Á™óÂè£  
fig, ax = plt.subplots(2, 1, figsize=(10, 8))  
  
# Ê∏©Â∫¶Êõ≤Á∫ø  
ax[0].set_title('Temperature (¬∞C)')  
ax[0].set_xlabel('Time (s)')  
ax[0].set_ylabel('Temperature (¬∞C)')  
ax[0].grid(True)  # Ê∑ªÂä†ÁΩëÊ†º  
line_temp, = ax[0].plot([], [], label='Temperature', color='r')  
  
# ÊπøÂ∫¶Êõ≤Á∫ø  
ax[1].set_title('Humidity (%)')  
ax[1].set_xlabel('Time (s)')  
ax[1].set_ylabel('Humidity (%)')  
ax[1].grid(True)  # Ê∑ªÂä†ÁΩëÊ†º  
line_hum, = ax[1].plot([], [], label='Humidity', color='b')  
  
# ËÆæÂÆöÂàùÂßã X ËΩ¥ËåÉÂõ¥  
ax[0].set_xlim(0, 60)  
ax[1].set_xlim(0, 60)  
  
# **Êõ¥Êñ∞Êï∞ÊçÆÂπ∂ÁªòÂà∂**  
def update_data(frame):  
    try:  
        # ‰ªéESP32Êé•Êî∂Êï∞ÊçÆ  
        data = s.recv(1024).decode().strip()  
        if data:  
            # Ëß£ÊûêÊ∏©Â∫¶ÂíåÊπøÂ∫¶  
            temp, hum = data.split(',')  
            temp = float(temp)  
            hum = float(hum)  
  
            # ËÆ°ÁÆóÂΩìÂâçÊó∂Èó¥Êà≥ÔºàÂÅáËÆæÊØèÊ¨°Êõ¥Êñ∞Èó¥Èöî 2sÔºâ  
            if times:  
                new_time = times[-1] + 2  # ÁªßÁª≠Â¢ûÂä† 2s Êó∂Èó¥  
            else:  
                new_time = 0  # Á¨¨‰∏ÄÂ∏ß‰ªé 0 ÂºÄÂßã  
  
            # Êõ¥Êñ∞Êï∞ÊçÆ  
            times.append(new_time)  
            temps.append(temp)  
            hums.append(hum)  
  
            # Êõ¥Êñ∞Êõ≤Á∫øÊï∞ÊçÆ  
            line_temp.set_data(times, temps)  
            line_hum.set_data(times, hums)  
  
            # **Êõ¥Êñ∞ X ËΩ¥ËåÉÂõ¥ÔºàÈò≤Ê≠¢ `min == max` Êä•ÈîôÔºâ**  
            if len(times) > 1:  
                ax[0].set_xlim(min(times), max(times))  
                ax[1].set_xlim(min(times), max(times))  
            else:  
                ax[0].set_xlim(new_time, new_time + 1)  
                ax[1].set_xlim(new_time, new_time + 1)  
  
            # **Âä®ÊÄÅÊõ¥Êñ∞ Y ËΩ¥ËåÉÂõ¥**  
            if temps:  
                ymin_t, ymax_t = min(temps), max(temps)  
                ax[0].set_ylim(ymin_t - 2, ymax_t + 2)  # Ê∏©Â∫¶ Y ËΩ¥Â¢ûÂä† 2¬∞C ÁºìÂÜ≤Âå∫  
  
            if hums:  
                ymin_h, ymax_h = min(hums), max(hums)  
                ax[1].set_ylim(ymin_h - 5, ymax_h + 5)  # ÊπøÂ∫¶ Y ËΩ¥Â¢ûÂä† 5% ÁºìÂÜ≤Âå∫  
  
        return line_temp, line_hum  
  
    except Exception as e:  
        print(f"Error: {e}")  
        return line_temp, line_hum  
  
# Âä®ÁîªÊõ¥Êñ∞ÔºåÈÅøÂÖçÁºìÂ≠òÈóÆÈ¢ò  
ani = animation.FuncAnimation(fig, update_data, interval=2000, cache_frame_data=False)
