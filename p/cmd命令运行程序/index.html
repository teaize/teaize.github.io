REM 定义菜单选项及相应的命令
set menuItems[1]=打开QQ
set commands[1]=E:\001-Sorfware\05-QQ\Bin\qq.exe
set menuItems[2]=执行微信
set commands[2]=E:\001-Sorfware\020-Wechart\WeChat\WeChat.exe
set menuItems[3]=打开知云
set commands[3]=E:\001-Sorfware\014-知云翻译\01\ZhiYunTranslator.exe
set menuItems[4]=退出程序

:menu
echo.
echo 请选择要执行的操作：
for /l %%i in (1,1,4) do echo %%i. !menuItems[%%i]!
set /p choice=请输入数字并按 Enter 键：

if "%choice%"=="4" goto op4
if "%choice%"=="3" goto op3
if "%choice%"=="2" goto op2
if "%choice%"=="1" goto op1

echo.
echo 请输入有效的数字！
echo.
pause
goto menu

:op1
echo.
echo 正在打开QQ...
echo.
echo 检查QQ是否在运行...

tasklist /FI "IMAGENAME eq qq.exe" 2>NUL | find /I /N "qq.exe">NUL

if "%ERRORLEVEL%"=="0" goto already_running_qq

echo.
echo 正在自动运行QQ...
echo.

start "" "!commands[1]!"

goto end_qq

:already_running_qq
echo.
echo QQ程序已在运行
echo.

pause
goto menu

:end_qq
echo.
echo 操作1的代码已执行完毕。
echo.

pause
goto menu

:op2
echo.
echo 检查WeChat是否在运行...
echo.

tasklist /FI "IMAGENAME eq WeChat.exe" 2>NUL | find /I /N "WeChat.exe">NUL

if "%ERRORLEVEL%"=="0" goto already_running_wechat

echo.
echo 正在自动运行微信...
echo.

start "" "!commands[2]!"

goto end_wechat

:already_running_wechat
echo.
echo WeChat程序已在运行
echo.

pause
goto menu

:end_wechat
echo.
echo 操作2的代码已执行完毕。
echo.

pause
goto menu

:op3
echo.
echo 正在打开知云...
echo.
echo 检查知云是否在运行...

tasklist /FI "IMAGENAME eq ZhiYunTranslator.exe" 2>NUL | find /I /N "ZhiYunTranslator.exe">NUL

if "%ERRORLEVEL%"=="0" goto already_running_zhiyun

echo.
echo 正在自动运行知云...
echo.

start "" "!commands[3]!"

goto end_zhiyun

:already_running_zhiyun
echo.
echo Zhiyun程序已在运行
echo.

pause
goto menu

:end_zhiyun
echo.
echo 操作3的代码已执行完毕。
echo.

pause
goto menu

:op4
echo.
echo 正在退出程序...
echo.

pause
exit
