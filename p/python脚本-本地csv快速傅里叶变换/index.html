import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from scipy.fft import fft, fftfreq
from scipy.signal import find_peaks

# ==========================================
# ã€é…ç½®åŒºåŸŸã€‘è¯·ä¿®æ”¹è¿™é‡Œçš„ä¿¡æ¯
# ==========================================
# 1. æ–‡ä»¶è·¯å¾„ (æ³¨æ„ï¼šWindowsè·¯å¾„è¯·ç”¨åæ–œæ  / æˆ– åŒæ–œæ  \\)
FILE_PATH = 'sensor_data.csv'  # ä¸¾ä¾‹: 'D:/data/test_01.xlsx'

# 2. ä½ è¦åˆ†æçš„é‚£ä¸€åˆ—çš„è¡¨å¤´åç§° (Excelç¬¬ä¸€è¡Œçš„åå­—)

# 3. é‡‡æ ·ç‡ (Hz): æ¯ç§’é‡‡é›†å¤šå°‘ä¸ªç‚¹ï¼Ÿ
# è­¦å‘Šï¼šè¿™ä¸ªæ•°å­—é”™äº†ï¼Œåˆ†æå‡ºæ¥çš„é¢‘ç‡å…¨éƒ½æ˜¯é”™çš„ï¼
FS = 1000.0                    # ä¸¾ä¾‹: 100Hz, 1000Hz

# ==========================================
# ä»£ç é€»è¾‘å¼€å§‹
# ==========================================

# è®¾ç½®ç»˜å›¾é£æ ¼å’Œå­—ä½“
plt.rcParams['font.sans-serif'] = ['SimHei']
plt.rcParams['axes.unicode_minus'] = False
plt.style.use('bmh') # ä½¿ç”¨ä¸€ç§æ¯”è¾ƒä¸“ä¸šçš„ç§‘ç ”ç»˜å›¾é£æ ¼

def load_and_analyze():
    # --- 1. è¯»å–æ•°æ® ---
    try:
        print(f"æ­£åœ¨è¯»å–æ–‡ä»¶: {FILE_PATH} ...")
        if FILE_PATH.endswith('.csv'):
            df = pd.read_csv(FILE_PATH)
        elif FILE_PATH.endswith(('.xls', '.xlsx')):
            df = pd.read_excel(FILE_PATH)
        else:
            print("é”™è¯¯ï¼šåªæ”¯æŒ .csv æˆ– .xlsx æ ¼å¼æ–‡ä»¶")
            return
            
        # æ£€æŸ¥åˆ—åæ˜¯å¦å­˜åœ¨
        if COLUMN_NAME not in df.columns:
            print(f"é”™è¯¯ï¼šåœ¨æ–‡ä»¶ä¸­æ‰¾ä¸åˆ°åˆ—å '{COLUMN_NAME}'")
            print(f"æ–‡ä»¶åŒ…å«çš„åˆ—åæœ‰: {list(df.columns)}")
            return
            
        # æå–æ•°æ®å¹¶è½¬ä¸º numpy æ•°ç»„
        # dropna() æ˜¯ä¸ºäº†é˜²æ­¢ç©ºå€¼æŠ¥é”™
        data_raw = df[COLUMN_NAME].dropna().values
        print(f"æˆåŠŸè¯»å– {len(data_raw)} ä¸ªæ•°æ®ç‚¹")

    except FileNotFoundError:
        print("------------------------------------------------------")
        print(f"ã€æ²¡æ‰¾åˆ°æ–‡ä»¶ã€‘: {FILE_PATH}")
        print("ä¸ºäº†æ¼”ç¤ºæ•ˆæœï¼Œæˆ‘å°†è‡ªåŠ¨ç”Ÿæˆä¸€æ®µå¸¦ 50Hz å¹²æ‰°çš„æ¨¡æ‹Ÿæ•°æ®...")
        print("------------------------------------------------------")
        # ç”Ÿæˆæ¨¡æ‹Ÿæ•°æ® (å¦‚æœæ²¡æ‰¾åˆ°æ–‡ä»¶)
        N_sim = 2000
        t_sim = np.linspace(0, N_sim/FS, N_sim)
        # çœŸå®ä¿¡å·(2Hz) + å·¥é¢‘å¹²æ‰°(50Hz) + éšæœºå™ªå£°
        data_raw = 2.0 + 0.5*np.sin(2*np.pi*2*t_sim) + 0.2*np.sin(2*np.pi*50*t_sim) + np.random.normal(0, 0.1, N_sim)

    # --- 2. é¢„å¤„ç† (å»ç›´æµ/å»åç§») ---
    # FFT å¯¹ç›´æµåˆ†é‡(0Hz)éå¸¸æ•æ„Ÿï¼Œå¦‚æœä¸å‡å»å¹³å‡å€¼ï¼Œ0Hzçš„æŸ±å­ä¼šé€šå¤©é«˜ï¼Œ
    # å¯¼è‡´ä½ çœ‹ä¸æ¸…åé¢çš„å¹²æ‰°ã€‚
    data_detrend = data_raw - np.mean(data_raw)

    # --- 3. æ‰§è¡Œ FFT ---
    N = len(data_detrend)
    yf = fft(data_detrend)
    xf = fftfreq(N, 1 / FS)

    # å–å‰åŠéƒ¨åˆ† (æ­£é¢‘ç‡)
    x_plot = xf[:N//2]
    y_plot = np.abs(yf[:N//2]) * 2 / N

    # --- 4. è‡ªåŠ¨æŠ“é¬¼ (å¯»æ‰¾å³°å€¼) ---
    # å¯»æ‰¾é¢‘è°±ä¸­æ˜¾è‘—çš„å°–å³° (é«˜åº¦è¶…è¿‡æœ€å¤§å€¼çš„ 10%)
    peaks, _ = find_peaks(y_plot, height=np.max(y_plot)*0.1)
    
    print("\nã€è¯Šæ–­æŠ¥å‘Šã€‘")
    print("-" * 30)
    print("å‘ç°æ½œåœ¨å¹²æ‰°æº (èƒ½é‡Top 5):")
    # å¯¹å³°å€¼æŒ‰é«˜åº¦æ’åº
    peak_indices = peaks[np.argsort(y_plot[peaks])[::-1]][:5] 
    
    for i in peak_indices:
        freq_val = x_plot[i]
        amp_val = y_plot[i]
        # å¿½ç•¥æ¥è¿‘ 0Hz çš„ä½é¢‘ (é‚£æ˜¯ä¿¡å·æœ¬èº«)
        if freq_val > 1.0: 
            print(f"-> é¢‘ç‡: {freq_val:.1f} Hz  (å¼ºåº¦: {amp_val:.4f})")
            if 48 < freq_val < 52:
                print("   [è­¦æŠ¥] ç–‘ä¼¼å·¥é¢‘å¹²æ‰° (ç”µæº/æ¥åœ°é—®é¢˜)")
            elif 98 < freq_val < 102:
                print("   [è­¦æŠ¥] ç–‘ä¼¼æ•´æµçº¹æ³¢ (ç”µæºæ¿æ»¤æ³¢ç”µå®¹é—®é¢˜)")

    # --- 5. ç»˜å›¾ ---
    fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(12, 10))

    # å›¾1: æ—¶åŸŸ (åŸå§‹æ³¢å½¢)
    # ä¸ºäº†é˜²æ­¢æ•°æ®é‡å¤ªå¤§å¡æ­»ï¼Œåªç”»å‰ 1000 ä¸ªç‚¹é¢„è§ˆ
    preview_N = min(1000, N)
    time_axis = np.arange(preview_N) / FS
    ax1.plot(time_axis, data_raw[:preview_N], 'k-', linewidth=1)
    ax1.set_title(f"åŸå§‹ä¿¡å·æ³¢å½¢ (ä»…æ˜¾ç¤ºå‰ {preview_N} ç‚¹)", fontsize=14)
    ax1.set_xlabel("æ—¶é—´ (s)")
    ax1.set_ylabel("å¹…å€¼")

    # å›¾2: é¢‘åŸŸ (FFT ç»“æœ)
    ax2.plot(x_plot, y_plot, 'r-', linewidth=1)
    # æ ‡è®°å‡ºæ‰¾åˆ°çš„å¹²æ‰°å³°
    ax2.plot(x_plot[peaks], y_plot[peaks], 'bx', label='æ£€æµ‹åˆ°çš„å³°å€¼')
    
    ax2.set_title("é¢‘è°±åˆ†æ (FFT Spectrum) - å¹²æ‰°ä¾¦æŸ¥", fontsize=14)
    ax2.set_xlabel("é¢‘ç‡ (Hz)")
    ax2.set_ylabel("å¹²æ‰°å¼ºåº¦")
    ax2.set_xlim(0, FS/2) # åªæ˜¾ç¤ºåˆ°å¥ˆå¥æ–¯ç‰¹é¢‘ç‡
    ax2.legend()

if __name__ == "__main__":
